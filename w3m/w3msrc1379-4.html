Compile Latest Version

1. Download wine and extract winexxx.tar.gz and enter in directory
2. sudo apt-get build-dep wine
3. sudo apt-get install ibosmesa6-dev
3. ./configure --help
4. ./configure --prefix=/home/caio/usr/  --exec-prefix=/home/caio/usr --with-x + other-options
5. make -j4 && make install
6. sudo ln -s /home/caio/usr/bin/wine   /bin/wine1.5   # create a shortcout wine1.5
